class collecArtiste extends SplHeap{

    
    public function __construct($tous) {
        $this->chargerArtistes($tous);
    }
    
    private function chargerArtistes($tous){
        $connexion = dbConnect::getInstance();
        if($tous != '-1'){
             $listeArt = $connexion->ExecuteSelect('Select * from artiste where numcont = ' . $tous);
        }
        else
        {    
            $listeArt = $connexion->ExecuteSelect('Select * from artiste order by numcont;');
        }
        
        foreach ($listeArt as $Artiste) {
            $Art = new artiste($Artiste['NUMCONT'],$Artiste['BIO'],$Artiste['DATENAISS'],$Artiste['DATEDECE'],$Artiste['PHOTO'],$Artiste['PSEUDO'],$Artiste['NOMCONT'],$Artiste['PRENOMCONT'],$Artiste['TELPORTCONT'],$Artiste['TELFIXECONT'],$Artiste['EMAILCONT'],$Artiste['WEBCONT']);
            $this->insert($Art);
        }
    }
    
    public function getFirstArt(){
        $this->top();
        return $this->current();
    }
    
     public function compare($artiste1, $artiste2)
    {
        return strcmp($artiste2->getNom(), $artiste1->getNom());
    }
}
